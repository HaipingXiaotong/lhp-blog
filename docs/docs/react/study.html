function SelfReducer(state = states, action: any): object {
    const { type, info } = action
    switch (type) {
        case SELF_USER:
            return {
                ...state,
                self: info
            }
        default:
            return state
    }
}
const store = createStore(SelfReducer)

上面的states是一个默认状态，action是dispatch的

export function changeSelf(info: object){
    return {
        info,
        type: SELF_USER
    }
}
